def days_to_burn_all_trees(grid):
    rows = len(grid)
    cols = len(grid[0])
    burned = false
    
    queue = []          # will store (row, col)
    healthy = 0
    days = 0

    # Step 1: collect all burning trees and count healthy ones
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == 2:
                burned = true
                queue.append((r, c))
            elif grid[r][c] == 1:
                healthy += 1

    # if no healthy trees, no time needed
    if healthy == 0:
        return 0
    if !burned
        return -1

    directions = [(1,0), (-1,0), (0,1), (0,-1)]

    # Step 2: BFS spread
    while queue:
        size = len(queue)
        burned_this_round = False

        for i in range(size):
            r, c = queue.pop(0)   # dequeue

            for dr, dc in directions:
                nr = r + dr
                nc = c + dc

                # check bounds and if tree is healthy
                if 0 <= nr < rows and 0 <= nc < cols:
                    if grid[nr][nc] == 1:
                        grid[nr][nc] = 2
                        queue.append((nr, nc))
                        healthy -= 1
                        burned_this_round = True

        if burned_this_round:
            days += 1

    # Step 3: check if any healthy trees remain
    if healthy > 0:
        return -1

    return days
